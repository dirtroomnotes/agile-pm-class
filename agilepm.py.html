def print_room_status(current_room, inventory, collected_artifacts, visited_rooms):
    """Prints the current room description, inventory, and available exits."""
    room = rooms[current_room]
    # Filter only uncollected artifacts for hints
    remaining_artifacts = [rooms[r]['artifact'] for r in rooms.keys()
                           if rooms[r]['artifact'] and rooms[r]['artifact'] not in collected_artifacts]

    print("\n" + "=" * 55)
    print(f"üèõÔ∏è  Current Room: {current_room}")
    print(room['description'])
    print(f"\nüì¶ Artifacts Collected: {len(collected_artifacts)}/6")
    print(f"   Inventory: {', '.join(inventory) if inventory else 'Empty'}")

    if room['artifact'] and room['artifact'] not in collected_artifacts:
        print(f"‚ú® You sense an artifact here: {room['artifact']}")

    print("\nüö™ Possible directions:")
    if room['exits']:
        for direction, dest in room['exits'].items():
            print(f"- {direction.capitalize()} ‚Üí {dest}")
    else:
        print("There are no clear exits from this room.")

    if remaining_artifacts:
        print(f"\nüí° Artifacts still needed: {', '.join(remaining_artifacts)}")
    print("=" * 55)


def game():
    """Main game loop and logic."""
    print("üèóÔ∏è Welcome to the Agile PM Adventure Game!")
    print("Your mission: collect all 6 project artifacts before meeting the Chaos Imp.\n")
    print("Commands:\nüëâ Move: go North | go South | go East | go West (or just type direction)")
    print("üëâ Quit: quit anytime to exit")
    print("-" * 55)

    current_room = 'War Room'
    inventory = []
    collected_artifacts = []
    visited_rooms = set(['War Room'])  # Track visited rooms (currently not used for display but good for game state)
    TOTAL_ARTIFACTS = 6

    while True:
        print_room_status(current_room, inventory, collected_artifacts, visited_rooms)

        # Automatically collect artifact upon entering/being in a room
        room_artifact = rooms[current_room]['artifact']
        if room_artifact and room_artifact not in collected_artifacts:
            collected_artifacts.append(room_artifact)
            inventory.append(room_artifact)
            print(
                f"\nü™Ñ You obtained: {room_artifact}! ({len(collected_artifacts)}/{TOTAL_ARTIFACTS} artifacts collected)")

        # Handle win/lose condition upon entering Cove of Chaos
        if current_room == 'Cove of Chaos':
            if len(collected_artifacts) == TOTAL_ARTIFACTS:
                print("\nüéâ You faced the Chaos Imp fully prepared and triumphed! YOU WIN!")
            else:
                print(
                    f"\nüíÄ You faced the Chaos Imp unprepared with only {len(collected_artifacts)}/{TOTAL_ARTIFACTS} artifacts... GAME OVER!")
            break

        move = input("\nEnter your move: ").strip().lower()
        if move == 'quit':
            print("Exiting the game. Goodbye!")
            break

        # Normalize move
        if move.startswith("go "):
            move = move[3:]

        # Check if the move is a valid exit
        if move in rooms[current_room]['exits']:
            current_room = rooms[current_room]['exits'][move]
            visited_rooms.add(current_room)
        else:
            print("‚ùå Invalid move. Try another direction (North, South, East, West).")


if __name__ == "__main__":
    game()
